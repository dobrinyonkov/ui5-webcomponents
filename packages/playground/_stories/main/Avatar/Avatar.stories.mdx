import { Story, Canvas, ArgsTable } from "@storybook/addon-docs";

import argTypes from "./argsTypes.js";
import Description from "./Description.md";

<Meta title="Components/Avatar" component="ui5-avatar" argTypes={argTypes} />

export const Template = (args) =>
  html`
    <ui5-avatar
      .aria-haspopup="${args.ariaHaspopup}"
      .color-scheme="${args.colorScheme}"
      .icon="${args.icon}"
      .initials="${args.initials}"
      ?interactive="${args.interactive}"
      .shape="${args.shape}"
      .size="${args.size}"
      @click="${args["ui5-click"]}"
      >${args.defaultSlot}</ui5-avatar
    >
  `;

export const StyledAvatar = (args) =>
  html` <ui5-avatar
    .shape="${args.shape}"
    .size="${args.size}"
    style="${args.style}"
    >${args.defaultSlot}</ui5-avatar
  >`;
